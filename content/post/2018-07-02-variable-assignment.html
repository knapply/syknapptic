C++

Pure C++

#include <iostream>

int main() {
  int var_CPP = 1;

  std::cout << var_CPP;
  
  return 0;
}

Rcpp

#include <Rcpp.h>       // R to/from C++ interface
// [[Rcpp::export]]

int RCPP_function() {
  int var_RCPP = 1;
  
  return var_RCPP;
}
RCPP_function() # access the C++ function from R like any other function
